  • There are three types of Statements →
    • Statement
    • PreparedStatement
    • Callable Statement
InterfacesDescription
StatementUseful when you are using static SQL statements at runtime. The Statement interface cannot accept parameters.
PreparedStatementUse this when you plan to use the SQL statements many times. The PreparedStatement interface accepts input parameters at runtime.
CallableStatementUse this when you want to access the database stored procedures. The CallableStatement interface can also accept runtime input parameters.

DriverManager

  • It acts as an bridge between user and drivers.
  • Methods →
    • registerDriver()
    • deRegisterDriver()
    • getConnection() etc ...

    EXAMPLE_1 →

    package JDBC.JDBC;

import java.sql.*;

public class App {
  public static void main(String[] args) {
    try {
    	Driver drv = new com.mysql.cj.jdbc.Driver;
      DriverManager.registerDriver(drv);
      
    	String url = "jdbc:mysql://localhost:3306/students";
    	String username = "root";
    	String password = "Shiddat@2104";
    	Connection con = DriverManager.getConnection(url, username, password);
    	
    	if(con!=null)  
        System.out.println("Connected"); 
	    else
        System.out.println("Not Connected");  	
    	con.close();
    }
    catch(Exception e) {
    	System.out.println(e);
    }
  }
}

PreparedStatement

  • This is the sub-interface of Statement
  • This Interface too provides methods to execute query with database but this interface accepts input parameters at runtime.
  • Methods →
    • setInt()
    • setString()
    • setDouble()
    • setFloat()
    • executeQuery()
    • executeUpdate() etc ...

    EXAMPLE_1 → (Insert into table)

    package JDBC.JDBC;

import java.sql.*;

public class App {
  public static void main(String[] args) {
    try {
    	Class.forName("com.mysql.cj.jdbc.Driver");
    	
    	String url = "jdbc:mysql://localhost:3306/students";
    	String username = "root";
    	String password = "Shiddat@2104";
    	Connection con = DriverManager.getConnection(url, username, password);
    	
    	PreparedStatement ps = con.prepareStatement("insert into student values(?,?)");
	     ps.setInt(1,31708);
	     ps.setString(2,"Abdul Katika");
	     ps.executeUpdate();
    		
    	con.close();
    }
    catch(Exception e) {
    	System.out.println(e);
    }
  }
}

Result Set

  • After retrieveing data from Database using statements, the data is stored in ResultSet interface.
  • ResultSet is an interface which stores data of tables.
  • It maintains a cursor pointing, which initially points to head of the tuple.
  • Methods →
    • next()
    • previous()
    • first()
    • last()
    • getInt()
    • getString() etc ...

    EXAMPLE_1 → (Retrieve Data from Database Table)

    package JDBC.JDBC;

import java.sql.*;

public class App {
  public static void main(String[] args) {
    try {
    	Class.forName("com.mysql.cj.jdbc.Driver");
    	
    	String url = "jdbc:mysql://localhost:3306/students";
    	String username = "root";
    	String password = "Shiddat@2104";
    	Connection con = DriverManager.getConnection(url, username, password);
    	
    	Statement stmt = con.createStatement();
    	ResultSet rs = stmt.executeQuery("select * from rohith");
    	
    	while(rs.next()) {
    		System.out.println(rs.getInt(1) + " : " + rs.getString(2));
    	}
    	
    	con.close();
    }
    catch(Exception e) {
    	System.out.println(e);
    }
  }
}

ResultSetMetaData

  • Methods →

    • getColumnName()
    • getColumnTypeName()
    • getTableName()

    EXAMPLE_1 → (Retrieve Data from Database Table)

import java.sql.*;

public class App {
  public static void main(String[] args) {

    	Class.forName("com.mysql.cj.jdbc.Driver");

    	String username = "root";
    	String password = "Shiddat@2104";
    	Connection con = DriverManager.getConnection(url, username, password);
    	
    	String s1 = "select * from student"; 
			PreparedStatement ps = con.prepareStatement(s1);
			ResultSet rs = ps.executeQuuery();
			ResultSetMetaData rsmd = rs.getMetaData();
			
			System.out.printnln("Total no. of columns: " +rsmd.getColumnCount());
			System.out.printnln("Column Type (1st Type): " +rsmd.getColumnTypeName(1));
			System.out.printnln("Column Name (1st Type): " +rsmd.getColumnName(1));
    	
    	con.close();

CallableStatement

  • This interface provides methods which are used to call stored procedures and functions.
    • Stored procedure → Procedure is defined in database manually.

    EXAMPLE_1 →

    CREATE OR REPLACE PROCEDURE "INSERT" (id in NUMBER, name in VARCHAR) IS
BEGIN
	insert into student values(id, name);
END;
    package JDBC.JDBC;

import java.sql.*;

public class App {
  public static void main(String[] args) {
    try {
    	Class.forName("com.mysql.cj.jdbc.Driver");
    	
    	String url = "jdbc:mysql://localhost:3306/students";
    	String username = "root";
    	String password = "Shiddat@2104";
    	Connection con = DriverManager.getConnection(url, username, password);
    	
    	CallableStatement cs = con.preparecall("{call INSERT(?,?)}");
    	cs.setInt(1, 10138);
    	cs.setString(2, "Ashwin");
    	cs.execute();
    	
    	con.close();
    }
    catch(Exception e) {
    	System.out.println(e);
    }
  }
}

Blob

  • Blob is used to retrieve an image from database.
    • Store →
      • Class - FileInputStream
      • Method - setBinaryStream()
    • Retrieve
      • Class - FileOutputStream
      • Method - getByte()

    EXAMPLE_1 → (Store an image)

    import java.sql.*;
import java.util.*;

class ImageTable {
	public static void main(String args[ ]) throws Exception {
		Class.forName("oracle.jdbc.driver.OracleDriver");
		Connection con = DriverManager.getConnection("jdbc:oracle:thin:@localhost:1521:xe","SYSTEM","admin");
		
		String s1 = "insert into imgtab values(?,?,?)";
    PreparedStatement ps = con.prepareStatement(s1);
		ps.setInt(1,100);
    ps.setString(2,"ABC");

    FileInputStream fis = new FileInpuitStream("path to the image");
    ps.setBinaryStream(3,fis,fis.available());
		ps.executeUpdate();

	  con.close();
   }
}

    EXAMPLE_2 → (Retrieve Image from DB)

    import java.sql.*;
import java.util.*;

class ImageTable {
	public static void main(String args[ ]) throws Exception {
		Class.forName("oracle.jdbc.driver.OracleDriver");
		Connection con = DriverManager.getConnection("jdbc:oracle:thin:@localhost:1521:xe","SYSTEM","admin");
		
		String s1 = "select * from imgtab";
    PreparedStatement ps = con.prepareStatement(s1);
		ResultSet rs = ps.executeQuery();

		while(rs.next()) {
		  Blob b = rs.getBlob(3);
			FileOutputStream fos = new FileOutputStream("path to store the image");
		  byte barr[] = b.getBytes(1, (int) b.length());
			fos.write(barr);
			fos.close();
	  }

	  con.close();
   }
}

Clob

  • Clob is used to retrieve an file from database.
    • Store →
      • Class - FileReader
      • Method - setCharacterStream()
    • Retrieve
      • Class - FileWriter
      • Method - getCharacterStream()

    EXAMPLE_1 → (Store an file)

import java.util.*;

    PreparedStatement ps = con.prepareStatement(s1);
		ps.setInt(1,100);
